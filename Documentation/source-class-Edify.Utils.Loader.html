 11:  * include each and every class on every script.
 12:  * 
 13:  * This allows you to not worry about including a class as long as the class file 
 14:  * follows the Standard for namespaces <vender>/path/of/class.php
 15:  * 
 16:  * @author IrishAdo <irishado@php-programmers.net>
 17:  * @licence Freetard - do what you wish technology, give back, keep, sell up to you.
 18:  */
 19: class Loader {
 20: 
 21:     // define a instance variable
 22:     /**
 23:      *
 24:      * @var SELF the current instance
 25:      */
 26:     private static $instance;
 27: 
 28:     /**
 29:      * @static
 30:      * @var Array list of vendors that the autoloader knows about.
 31:      */
 32:     private static $vendors;
 33: 
 34:     /** initalise the class and add the Path to this Library Vendor
 35:      * 
 36:      * @author IrishAdo <irishado@php-programmers.net>
 37:      */
 38:     public static function init() {
 39:         if (self::$instance == NULL) {
 40:             self::$instance = new self();
 41:         }
 42:         
 43:         self::registerVendor('Edify', dirname(dirname(__FILE__)));
 44:         return self::$instance;
 45:     }
 46: 
 47:     /** Constructor which sets the loader function to be an auto loader
 48:      *
 49:      * @author IrishAdo <irishado@php-programmers.net>
 50:      */
 51:     public function __construct() {
 52:         spl_autoload_register(array($this, "auto"));
 53:     }
 54: 
 55:     /** Preprocessor function which will attempt to include a file contains a class before the new command executes
 56:      * 
 57:      * @author IrishAdo <irishado@php-programmers.net>
 58:      * @param Class name 
 59:      */
 60:     public function auto($class) {
 61:         // split the namespace and class up into parts
 62:         $parts = explode("\\", $class);
 63: 
 64:         // the first part of the namespace is the vendor a unique project identifier
 65:         $choosenVendor = $parts[0];
 66: 
 67:         // remove the end and the beginning of the path
 68:         unset($parts[0]);
 69: 
 70:         $path = implode("/", $parts);
 71: 
 72:         // if we have not listed the path for that vendor then we don't know where it is so die.
 73:         if (!isset(self::$vendors[$choosenVendor])) {
 74:             error_log("[Edify\Utils\Loader] Vendor $choosenVendor not found use \Edify\Utils\Loader::AddVendorPath(\$vendor,\$path);");
 75:             throw new Exception('Sorry we could not find a class located at '. $checkPathforFile);
 76:         } else {
 77: 
 78:             // work out the path to look for the file based on the Vendors path.
 79:             $checkPathforFile = realpath(self::$vendors[$choosenVendor]) . "/" . $path . ".php";
 80: 
 81:             // if the file does exist then include it or die.
 82:             if (file_exists($checkPathforFile)) {
 83:                 require_once($checkPathforFile);
 84:             } else {
 85:                 error_log("[Edify\Utils\Loader] We could not find the following file in the specified vendor -> $checkPathforFile");
 86:                 throw new Exception('Sorry we could not find a class located at  '. $checkPathforFile);
 87:             }
 88:         }
 89:     }
 90: 
 91:     /** function to register a vendor to the Auto loader so that it knows where
 92:      * subdirectory the files for that vendor is located.
 93:      *
 94:      * For example if I created a new application called Mapper then the
 95:      * Namespace for the elements in Mapper should be Mapper\*  by registering
 96:      * the vendor "Mapper" we can look in the correct location for all of the
 97:      * Classes for that namespace.  This allows developers to have many different
 98:      * namespaces and for the Autoloader functionality to know where to go for each.
 99:      *
100:      * @author IrishAdo <irishado@php-programmers.net>
101:      * @param String the name of the Vendor
102:      * @param String Path to look for the class file.
103:      * @example self::AddVendorPath('Edify', PP_NET_CL_PATH);
104:      */
105:     public static function registerVendor($vendor, $path) {
106:         self::$vendors[$vendor] = $path;
107:     }
